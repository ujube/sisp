🔥 1. Network Proxy (tüm istekleri yakalama)

👉 Amaç:
Sandbox içindeki tüm HTTP/HTTPS isteklerini loglamak

utils/networkProxy.js

const http = require('http');

function startProxy(port = 8888) {
  const logs = [];

  const server = http.createServer((req, res) => {
    const log = {
      method: req.method,
      url: req.url,
      headers: req.headers,
      time: new Date().toISOString()
    };

    logs.push(log);

    console.log("🌐 Request:", req.method, req.url);

    res.writeHead(200);
    res.end("Blocked by Safe Install Proxy");
  });

  server.listen(port, () => {
    console.log(`🛰️ Proxy running on port ${port}`);
  });

  return { server, logs };
}

module.exports = { startProxy };

2. Child Process Hook (en güçlü feature)

👉 Amaç:
Sandbox içinde çalışan şüpheli komutları yakalamak

📁 sandbox/runner.js

const cp = require('child_process');

const originalSpawn = cp.spawn;

cp.spawn = function (...args) {
  const cmd = args[0];

  console.log("⚙️ Process:", cmd);

  if (cmd.includes('curl') || cmd.includes('wget') || cmd.includes('bash')) {
    console.log("🚨 Suspicious command detected:", cmd);
  }

  return originalSpawn.apply(this, args);
};

// npm install başlat
require('child_process').execSync('npm install', {
  stdio: 'inherit'
});


🐳 Dockerfile (runner ile)

FROM node:18
