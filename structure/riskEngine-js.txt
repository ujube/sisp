function calculateRisk(data) {
  let score = 0;

  if (data.hasPostinstall) score += 0.3;
  if (data.networkCalls > 0) score += 0.3;
  if (data.sensitiveFiles > 0) score += 0.2;
  if (data.obfuscation) score += 0.2;

  return score;
}

function decision(score) {
  if (score > 0.7) return "BLOCK";
  if (score > 0.4) return "WARN";
  return "ALLOW";
}

module.exports = { calculateRisk, decision };
