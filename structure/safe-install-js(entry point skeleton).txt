#!/usr/bin/env node

const { execSync } = require('child_process');

console.log("🛡️ Safe Install starting...");

// 1. Pre-scan
console.log("🔍 Scanning package.json...");
try {
  const pkg = require(process.cwd() + '/package.json');

  if (pkg.scripts?.postinstall) {
    console.log("⚠️ postinstall script detected");
  }
} catch (e) {
  console.log("No package.json found");
}

// 2. Sandbox run (placeholder)
console.log("🐳 Running in sandbox...");
try {
  execSync('docker --version', { stdio: 'ignore' });
} catch {
  console.log("❌ Docker not installed");
  process.exit(1);
}

// 3. Decision (mock)
const riskScore = 0.3;

if (riskScore > 0.7) {
  console.log("❌ Install blocked");
  process.exit(1);
}

// 4. Real install
console.log("✅ Safe → running npm install...");
execSync('npm install', { stdio: 'inherit' });
